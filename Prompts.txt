USERS TABLE

👉 Stores login + personal + role info

CREATE TABLE users (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    gender VARCHAR(10),

    email VARCHAR(100) UNIQUE NOT NULL,
    address VARCHAR(255),

    account_type VARCHAR(50),

    mobile_number VARCHAR(15) UNIQUE NOT NULL,
    pan_number VARCHAR(20) UNIQUE NOT NULL,
    aadhar_number VARCHAR(20) UNIQUE NOT NULL,

    username VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,

    role VARCHAR(20) DEFAULT 'CUSTOMER',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);


💳 3. ACCOUNTS TABLE (IMPORTANT)

👉 Real banking system uses separate account table

CREATE TABLE accounts (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    account_number VARCHAR(20) UNIQUE NOT NULL,
    balance DECIMAL(15,2) DEFAULT 0.00,

    user_id BIGINT,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id)
    REFERENCES users(id)
    ON DELETE CASCADE
);



💸 4. TRANSACTIONS TABLE
CREATE TABLE transactions (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,

    from_account VARCHAR(20),
    to_account VARCHAR(20),

    amount DECIMAL(15,2) NOT NULL,

    type VARCHAR(20),       -- TRANSFER / DEPOSIT / WITHDRAW
    status VARCHAR(20),     -- SUCCESS / FAILED

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
